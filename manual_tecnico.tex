\section{Arquitectura del Sistema}

\subsection{Visión General}
El sistema sigue una arquitectura cliente-servidor moderna:
\begin{itemize}
    \item \textbf{Frontend}: Aplicación web en React + TypeScript + Vite.
    \item \textbf{Backend}: API REST en Python con FastAPI.
\end{itemize}
La comunicación se realiza mediante HTTP/JSON a través de una API RESTful en el puerto 8003.

\subsection{Tecnologías del Frontend}
\begin{itemize}
    \item React 18: Framework de interfaz de usuario.
    \item TypeScript: Lenguaje con tipos estáticos.
    \item Vite: Herramienta de build y desarrollo.
    \item React Router DOM: Navegación entre páginas.
    \item Axios: Cliente HTTP para llamadas a la API.
    \item Lucide React: Biblioteca de iconos.
\end{itemize}

\subsection{Tecnologías del Backend}
\begin{itemize}
    \item FastAPI: Framework web moderno de Python.
    \item Uvicorn: Servidor ASGI de alto rendimiento.
    \item Pydantic: Validación de datos.
    \item SQLite: Base de datos embebida.
    \item sqlite-vec: Extensión para búsquedas vectoriales (opcional).
\end{itemize}

\section{Frontend}

\subsection{Estructura del Proyecto}
\begin{verbatim}
frontend/
├── src/
│   ├── components/   # Navbar, Footer
│   ├── pages/        # Home, About, Learning, Profile, Autenticación
│   ├── services/     # API calls
│   ├── hooks/        # Custom hooks
│   ├── types/        # TypeScript types
│   └── styles/       # Estilos globales
├── public/           # Recursos estáticos
└── package.json      # Dependencias
\end{verbatim}

\subsection{Páginas Principales}
\begin{itemize}
    \item \textbf{Home}: Página de bienvenida con información del proyecto.
    \item \textbf{About}: Información del equipo y del proyecto.
    \item \textbf{Learning}: Interfaz de lecciones con editor de código integrado.
    \item \textbf{Profile}: Estadísticas de progreso del usuario.
    \item \textbf{Diagnóstico}: Evaluación inicial para determinar el lenguaje recomendado.
    \item \textbf{ExpertAnnotations}: Panel para expertos.
\end{itemize}

\subsection{Ejecución del Frontend}
\begin{lstlisting}[language=bash]
# Instalar dependencias
bun install

# Iniciar servidor de desarrollo
bun run dev

# Construir para producción
bun run build
\end{lstlisting}

\subsection{Estructura del Proyecto}

\subsection{Endpoints de la API}

\textbf{Autenticación:}
\begin{itemize}
    \item POST /api/v1/login - Iniciar sesión
    \item POST /api/v1/register - Registrar usuario
    \item GET /api/v1/hc - Health check
\end{itemize}

\textbf{Diagnóstico:}
\begin{itemize}
    \item GET /api/v1/questions - Obtener preguntas
    \item POST /api/v1/diagnostic-result - Guardar resultado
    \item GET /api/v1/diagnostic-results/\{username\} - Ver resultados
\end{itemize}

\textbf{Contenido Educativo:}
\begin{itemize}
    \item GET /api/v1/content/user/\{username\} - Contenido personalizado
    \item GET /api/v1/content/\{language\} - Contenido por lenguaje
    \item GET /api/v1/content/\{language\}/\{level\} - Contenido por nivel
\end{itemize}

\textbf{Ejecución de Código:}
\begin{itemize}
    \item POST /api/v1/execute - Ejecutar código (Python, C++, Java)
\end{itemize}

\textbf{Panel de Experto:}
\begin{itemize}
    \item GET /api/v1/expert/students - Listar estudiantes
    \item POST /api/v1/expert/annotation - Crear anotación
    \item PUT /api/v1/expert/annotation/\{id\} - Actualizar anotación
    \item DELETE /api/v1/expert/annotation/\{id\} - Eliminar anotación
\end{itemize}

\section{Base de Datos}

\subsection{Tablas Principales}

\textbf{users:}
\begin{verbatim}
id, username, password (hash SHA256), role, created
\end{verbatim}

\textbf{questions:}
\begin{verbatim}
id, number, text, language, yes_next, no_next
\end{verbatim}

\textbf{diagnostic\_results:}
\begin{verbatim}
id, username, result, python_score, cpp_score, java_score, 
difficulty_responses (JSON), created
\end{verbatim}

\textbf{temas:}
\begin{verbatim}
id, curso, nivel, id_capitulo, titulo_capitulo
\end{verbatim}

\textbf{secciones:}
\begin{verbatim}
id, tema_id, id_seccion, tipo, titulo, cuerpo, codigo
\end{verbatim}

\textbf{expert\_annotations:}
\begin{verbatim}
id, expert_username, student_username, diagnostic_id, 
annotation, created, updated
\end{verbatim}

\subsection{Contenido Educativo}
Archivos JSON en la raíz del proyecto:
\begin{itemize}
    \item py\_basico.json, py\_intermedio.json, py\_avanzado.json
    \item cpp\_basico.json, cpp\_intermedio.json, cpp\_avanzado.json
    \item java\_basico.json, java\_intermedio.json, java\_avanzado.json
\end{itemize}

Estructura de cada archivo:
\begin{verbatim}
{
  "curso": "Python",
  "nivel": "Básico",
  "total_capitulos": 5,
  "capitulos": [
    {
      "num_capitulo": 1,
      "titulo": "Introducción",
      "secciones": [
        {
          "id": 1,
          "tipo": "contenido",
          "titulo": "Título",
          "cuerpo": "Descripción...",
          "codigo": "print('Hola')"
        }
      ]
    }
  ]
}
\end{verbatim}

\section{Ejecución del Sistema}

\subsection{Instalación y Ejecución}
El archivo launch.py facilita la ejecución:

\begin{enumerate}
    \item Crea un entorno virtual si no existe
    \item Instala las dependencias necesarias
    \item Inicia el servidor FastAPI en el puerto 8003
\end{enumerate}

\begin{lstlisting}[language=bash]
python launch.py
\end{lstlisting}

El servidor estará disponible en:
\begin{itemize}
    \item API: http://localhost:8003
    \item Health check: http://localhost:8003/api/v1/hc
\end{itemize}
